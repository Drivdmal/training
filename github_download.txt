1. cd 				# перейти в текущую папку
2. git init 				#определить текущую директорию как директорию репозитория
3. git add . 			# добавляем файлы в локальный репозиторий, чтобы добавить файлы в первый коммит
4. git commit -m "First commit"	# коммит указанных изменений и подготовка к отправке в удаленный репозиторий
5. copy link from https:/......		# копируем ссылку на удаленном репозитории
6. git remote -v			# указание на новый удаленный репозиторий
7. git remote add origin ссылка 	# проверка нового удаленного репозитория
8. git pus origin master		# отправка изменений локального репозитория в удаленный, помеченный как источник
9. Enter token			# ввести временный токен из настроек на гитхабе
